    'FPR': list(fpr_knn3) + list(fpr_dt3),
    'TPR': list(tpr_knn3) + list(tpr_dt3),
    'Model': ['KNN (k=3)'] * len(fpr_knn3) + ['Decision Tree (max_depth=3)'] * len(fpr_dt3)
})

(
    ggplot(roc_df, aes(x='FPR', y='TPR', color='Model'))
